## Install

    npm install @dakan/ui

# Example

    import * as React from 'react'
    import {Button, ThemeProvider, navTheme} from '@dakan/ui'

    export default function App() {
        return (
            <ThemeProvider theme={navTheme()}>
                <Button />
            </ThemeProvider>
        );
    }
